```html
      <div class="cardmask" v-if="cardMaskShow">
        <transition name="cardpull">
          <div class="message-box-wrap" v-if="cardShow">
            <div class="message-box">
              <div class="message">
                <div
                  class="card"
                  v-for="(item, index) in leaveType"
                  :key="item"
                  @click="selectPop(index)"
                >{{item.typeName}}</div>
              </div>
              <div class="btn-group-2">
                <div class="cancel-btn" @click="closeCard">取消</div>
              </div>
            </div>
          </div>
        </transition>
      </div>
```
```javascript
cardMaskShow: false, 
cardShow: false, 

openCard(x) {
      this.popType = x;
      this.cardMaskShow = !this.cardMaskShow;
      setTimeout(() => {
        this.cardShow = !this.cardShow;
      }, 100);
    },
    closeCard() {
      this.cardShow = !this.cardShow;
      setTimeout(() => {
        this.cardMaskShow = !this.cardMaskShow;
      }, 200);
    },
```
